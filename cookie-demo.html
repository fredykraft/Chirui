💻 Console Commands for Analysis

Copy and paste these commands into your browser console (F12 → Console):

Basic Commands:

// Get all statistics ActivityTracker.getStats() // Check if tracking is enabled ActivityTracker.isEnabled() // Get specific visitor info CookieManager.getJSON('visitor_info') // Get current session data CookieManager.getJSON('visitor_session') // Get all cookies CookieManager.getAll()

Analysis Commands:

// Calculate average time per page var stats = ActivityTracker.getStats(); var avgTime = stats.totalTimeSpent / stats.pageViews.length; console.log('Average time per page:', avgTime, 'seconds'); // View most visited pages var pageCounters = CookieManager.getJSON('visitor_page_counters'); console.log('Page visit counts:', pageCounters); // Get recent activity summary var activities = CookieManager.getJSON('visitor_activities'); activities.forEach(function(act, i) { console.log(i+1, act.path, '→', act.timeSpent + 's', 'Scroll:', act.scrollDepth + '%'); }); // Track a test event ActivityTracker.trackEvent('demo_test', { action: 'testing_tracking', timestamp: new Date().toISOString() });

Data Export:

// Export all data as JSON var stats = ActivityTracker.getStats(); var jsonStr = JSON.stringify(stats, null, 2); console.log(jsonStr); // Download as file var blob = new Blob([jsonStr], {type: 'application/json'}); var url = URL.createObjectURL(blob); var a = document.createElement('a'); a.href = url; a.download = 'visitor-data-' + new Date().toISOString() + '.json'; a.click();
