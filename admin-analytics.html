★ Server-Side Integration Guide

To collect data from ALL visitors (not just your own browser), add this code to activity-tracker.js:

// Add to the saveCurrentPageData() function:
function saveCurrentPageData() {
  if (!config.enabled || !state.pageLoadTime) return;

  var timeSpent = Math.round((state.lastActivityTime - state.pageLoadTime) / 1000);
  
  var pageActivity = {
    path: window.location.pathname,
    url: window.location.href,
    title: document.title,
    sessionId: state.sessionId,
    visitorId: state.visitorId,
    timeSpent: timeSpent,
    scrollDepth: state.scrollDepth,
    clicks: state.clicks.length,
    clickDetails: state.clicks.slice(-10),
    timestamp: new Date().toISOString()
  };

  // Send to your server
  sendToServer(pageActivity);
  
  // ... rest of existing code
}

// Add this new function:
function sendToServer(data) {
  // Replace with your server endpoint
  var endpoint = '/api/track-activity';
  
  fetch(endpoint, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(data),
    keepalive: true // Ensures data is sent even if page is closing
  }).catch(function(error) {
    console.error('Analytics error:', error);
  });
}

Server Endpoint Examples:

  • External Services: Google Analytics, Mixpanel, Plausible
  • Self-hosted: Set up a simple Node.js/Python/PHP endpoint
  • Firebase: Use Firestore to store analytics data
  • Google Sheets: Use Google Apps Script as endpoint
